# STEP 1: REMOUNT THE EVIDENCE IMAGE
# (ewfmount disappears after shutdown)

sudo mkdir -p /mnt/ewf
sudo ewfmount /home/vboxuser/Evidance/usb_a.E01 /mnt/ewf
sudo ls -la /mnt/ewf/

# ‚úÖ You should see "ewf1" in the listing
#

#   sudo umount /mnt/ewf
#   sudo ewfmount /home/vboxuser/Evidance/usb_a.E01 /mnt/ewf
#
# ‚ùå If it says "No such file or directory":
#   Check spelling ‚Äî your folder is Evidance not Evidence
#

#   sudo apt install ewf-tools -y
